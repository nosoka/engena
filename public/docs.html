Authentication¶

Create new user
POST/api/auth/register

Validate if all the necessary information is provided. If all is well create a new user account and respond with an access token

// Sample Request
$.ajax({
    url: "https://engena.co.za/api/auth/register",
    type: "POST",
    data: {
        "username": "foo",
        "password": "bar",
        "fullname": "Gorrest Gump",
        "email": "abc@xyz.com",
        "mobile": 9876543210
    }
});

// access token is returned if WeChat-ID is present in headers.
    headers: { WeChat-ID: "xyz.." },

    url: "https://api.engena.co.za/api/reserves",
    type: "GET",
});

    url: "https://api.engena.co.za/api/reserves?activity=1,2,3",
    type: "GET",
});

// Get all reserves where (region=1 or 2).

    url: "https://api.engena.co.za/api/reserves?region=1,2",
    type: "GET",
});

// Get all reserves with (activity=3) and (region=2).
